<script>

  // ── CONFIG — credentials entered at runtime ──

  let SUPABASE_URL = '';

  let SUPABASE_KEY = '';


  function initDashboard() {

    const url = document.getElementById('configUrl').value.trim();

    const key = document.getElementById('configKey').value.trim();

    if (!url || !key) { alert('Please enter both your Supabase URL and key.'); return; }

    SUPABASE_URL = url;

    SUPABASE_KEY = key;

    document.getElementById('configOverlay').style.display = 'none';

    loadUsers();

  }


  let allUsers = [];

  let currentFilter = 'all';

  let currentUserId = null;

  let currentStatus = null;


  // ── FETCH USERS ──

  async function loadUsers() {

    document.getElementById('usersTable').innerHTML =

      '<tr class="loading-row"><td colspan="6">Loading...</td></tr>';

    try {

      const res = await fetch(`${SUPABASE_URL}/rest/v1/users?select=*&order=created_at.desc`, {

        headers: {

          'apikey': SUPABASE_KEY,

          'Authorization': `Bearer ${SUPABASE_KEY}`

        }

      });

      allUsers = await res.json();

      updateStats();

      renderTable();

      document.getElementById('lastUpdated').textContent =

        'Updated ' + new Date().toLocaleTimeString('en-AU', {hour:'2-digit',minute:'2-digit'});

    } catch(e) {

      document.getElementById('usersTable').innerHTML =

        '<tr class="loading-row"><td colspan="6">⚠️ Could not load users. Check your Supabase config.</td></tr>';

    }

  }


  // ── STATS ──

  function updateStats() {

    const pending  = allUsers.filter(u => u.status === 'pending').length;

    const reviewed = allUsers.filter(u => u.status === 'reviewed').length;

    const switched = allUsers.filter(u => u.status === 'switched').length;

    const weekAgo  = new Date(Date.now() - 7*24*60*60*1000);

    const thisWeek = allUsers.filter(u => new Date(u.created_at) > weekAgo).length;


    document.getElementById('stat-total').textContent    = allUsers.length;

    document.getElementById('stat-pending').textContent  = pending;

    document.getElementById('stat-reviewed').textContent = reviewed;

    document.getElementById('stat-switched').textContent = switched;

    document.getElementById('stat-week').textContent     = thisWeek;

  }


  // ── RENDER TABLE ──

  function renderTable() {

    const search = document.getElementById('searchInput').value.toLowerCase();

    let users = allUsers.filter(u => {

      const matchFilter = currentFilter === 'all' || u.status === currentFilter;

      const matchSearch = !search ||

        `${u.first_name} ${u.last_name}`.toLowerCase().includes(search) ||

        (u.email || '').toLowerCase().includes(search);

      return matchFilter && matchSearch;

    });


    if (users.length === 0) {

      document.getElementById('usersTable').innerHTML =

        `<tr><td colspan="6"><div class="empty-state"><div class="icon">📭</div><p>No users found</p></div></td></tr>`;

      return;

    }


    document.getElementById('usersTable').innerHTML = users.map(u => `

      <tr onclick="openModal(${u.id})">

        <td>

          <div class="td-name">${u.first_name || ''} ${u.middle_name || ''} ${u.last_name || ''}</div>

        </td>

        <td><div class="td-email">${u.email || '—'}</div></td>

        <td>${badgeHTML(u.status)}</td>

        <td><div class="td-date">${u.suburb || '—'}, ${u.state || ''}</div></td>

        <td><div class="td-date">${formatDate(u.created_at)}</div></td>

        <td><div class="td-date">${u.is_moving || '—'}</div></td>

      </tr>

    `).join('');

  }


  function badgeHTML(status) {

    const map = {

      pending:  ['badge-pending',  '● Pending'],

      reviewed: ['badge-reviewed', '● Reviewed'],

      switched: ['badge-switched', '● Switched'],

    };

    const [cls, label] = map[status] || ['badge-pending', '● Pending'];

    return `<span class="badge ${cls}">${label}</span>`;

  }


  function formatDate(iso) {

    if (!iso) return '—';

    return new Date(iso).toLocaleDateString('en-AU', {day:'numeric', month:'short', year:'numeric'});

  }


  // ── FILTER ──

  function setFilter(f, btn) {

    currentFilter = f;

    document.querySelectorAll('.filter-btn').forEach(b => b.classList.remove('active'));

    btn.classList.add('active');

    renderTable();

  }

  // ── MODAL ──

  function openModal(id) {

    const u = allUsers.find(u => u.id === id);

    if (!u) return;

    currentUserId = id;


    document.getElementById('modalName').textContent =

      `${u.first_name || ''} ${u.last_name || ''}`;

    document.getElementById('d-name').textContent =

      [u.first_name, u.middle_name, u.last_name].filter(Boolean).join(' ');

    document.getElementById('d-dob').textContent    = u.date_of_birth || '—';

    document.getElementById('d-email').textContent  = u.email || '—';

    document.getElementById('d-phone').textContent  = u.phone || '—';

    document.getElementById('d-address').textContent =

      [u.address1, u.address2].filter(Boolean).join(', ') || '—';

    document.getElementById('d-location').textContent =

      [u.suburb, u.state, u.postcode].filter(Boolean).join(' ') || '—';

    document.getElementById('d-moving').textContent    = u.is_moving || '—';

    document.getElementById('d-licence').textContent   = u.licence_number || '—';

    document.getElementById('d-licencecard').textContent = u.licence_card || '—';

    document.getElementById('d-licencestate').textContent = u.licence_state || '—';

    document.getElementById('d-consent').textContent   = u.consent_given ? '✓ Yes' : '✗ No';

    document.getElementById('modalNotes').value = u.notes || '';


    selectStatus(u.status || 'pending');

    document.getElementById('modalOverlay').classList.add('open');

  }


  function selectStatus(s) {

    currentStatus = s;

    ['pending','reviewed','switched'].forEach(opt => {

      const el = document.getElementById('opt-' + opt);

      el.className = 'status-opt';

      if (opt === s) el.classList.add('sel-' + s);

    });

  }


  function closeModal(e) {

    if (e.target === document.getElementById('modalOverlay')) closeModalDirect();

  }


  function closeModalDirect() {

    document.getElementById('modalOverlay').classList.remove('open');

  }


  // ── SAVE ──

  async function saveUser() {

    const notes = document.getElementById('modalNotes').value;

    try {

      await fetch(`${SUPABASE_URL}/rest/v1/users?id=eq.${currentUserId}`, {

        method: 'PATCH',

        headers: {

          'apikey': SUPABASE_KEY,

          'Authorization': `Bearer ${SUPABASE_KEY}`,

          'Content-Type': 'application/json',

          'Prefer': 'return=minimal'

        },

        body: JSON.stringify({ status: currentStatus, notes: notes })

      });


      // update local data

      const u = allUsers.find(u => u.id === currentUserId);

      if (u) { u.status = currentStatus; u.notes = notes; }


      updateStats();

      renderTable();

      closeModalDirect();

      showToast('✓ Saved!');

    } catch(e) {

      showToast('⚠️ Error saving');

    }

  }


  // ── TOAST ──

  function showToast(msg) {

    const t = document.getElementById('toast');

    t.textContent = msg;

    t.classList.add('show');

    setTimeout(() => t.classList.remove('show'), 2500);

  }
